  <script>
    const terminal = document.getElementById("terminal");
    const cmdInput = document.getElementById("cmdInput");

    const commands = {
      help: "Available commands: about, contact, books, clear, help",
      about: "Worldly Quill is an independent publishing brand focused on travel, storytelling, and cultural guides.",
      contact: "Email: contact@worldlyquill.com | IG: @worldlyquill",
      books: "Current title: 'Welcome to London' – Launching July 2025.",
      clear: "clear"
    };

    cmdInput.addEventListener("keydown", function (e) {
      if (e.key === "Enter") {
        const input = cmdInput.value.trim().toLowerCase();
        const line = document.createElement("div");
        line.className = "line";
        line.textContent = "> " + input;
        terminal.appendChild(line);

        const output = document.createElement("div");
        output.className = "line";

        if (commands[input]) {
          if (input === "clear") {
            terminal.innerHTML = "";
          } else {
            output.textContent = commands[input];
            terminal.appendChild(output);
          }
        } else if (input) {
          output.textContent = `'${input}' is not recognized as a command. Type 'help' for list.`;
          terminal.appendChild(output);
        }

        cmdInput.value = "";
        window.scrollTo(0, document.body.scrollHeight);
      }
    });
  </script>
